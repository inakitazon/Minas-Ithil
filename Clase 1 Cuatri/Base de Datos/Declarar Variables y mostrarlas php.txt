set @suspendidos=5;
set @num_suspensos=0;
set @aprobados=6;
set @num_aprobados=0;
set @sobresaliente=10;
set @num_sobresaliente=0;


-- suspendidos
SELECT * from reviews WHERE Valoracion<=@suspendidos;
SELECT COUNT(*) INTO @num_suspensos FROM reviews WHERE Valoracion<=@suspendidos;

-- aprobados
SELECT * from reviews WHERE Valoracion>=@aprobados;
SELECT COUNT(*) INTO @num_aprobados from reviews WHERE Valoracion>=@aprobados;

-- sobresaliente
SELECT * from reviews WHERE Valoracion=@sobresaliente;
SELECT COUNT(*) into @num_sobresalientes from reviews WHERE Valoracion=@sobresaliente;


SELECT @num_aprobados as aprobados,@num_suspensos as suspensos,@num_sobresaliente as sobresaliente;
