JMP PRIMER 

PERSONA1: DB "MAYOR"
	DB 0

PERSONA2: DB "MENOR"
	DB 0 

PRIMER:
	MOV A,8
	MOV B,6
	MOV D,232
	CMP A,B
	JA MAYOR
	JB MENOR

MAYOR:
	MOV B,0
	MOV C,PERSONA1
OPERACION1:
	MOV A,[C]
	MOV [D],A
	INC C
	INC D
	CMP B,[C]
	JNZ OPERACION1
	HLT

MENOR:
	MOV B,0
	MOV C,PERSONA2
OPERACION2:
	MOV A,[C]
	MOV [D],A
	INC C
	INC D
	CMP B,[C]
	JNZ OPERACION2
	HLT
