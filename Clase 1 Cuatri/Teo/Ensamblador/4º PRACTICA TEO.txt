EJERCICIO Nº6:


CODIFICAR CADENA DE CARACTERES Y DESCODIFICAR


JMP Prepare

String: DB "HOLA"
        DB 0

Prepare:

    MOV C, String
    MOV D,232
    MOV B,0

START:
   MOV A, [C]
   CMP B, A
   JZ FINISH
   SUB A,7
   MOV [D],A
   INC C
   INC D
   CMP B, A
   JNZ START
   HLT



FINISH:
 
   DEC D
   MOV A,[D]
   ADD A,7
   MOV [D],A
   CMP D,231
   JNZ FINISH
