JMP Prepare

N1: DB 0
N2: DB 9

Prepare:

  ADD A,48
  ADD B,48
  MOV D,232
  MOV [D], 48
  MOV C,[N2]
  ADD C,48
  INC D
  MOV [D],B

CONT:

  
  INC B
  MOV [D], B
  CMP B,C
  JNZ CONT
  CMP A,C
  JZ FIN
  MOV B,48
  MOV [D],B
  INC A

CONT2:
   
   DEC D
   MOV [D], A
   INC D
   CMP A,[N2]
   JNZ CONT

FIN:
 

 HLT
